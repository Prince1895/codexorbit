CSS Cheatsheet

Quick reference for common CSS properties

Selectors

Basic Selectors

/* Universal Selector */
* {
  margin: 0;
  padding: 0;
}

/* Element Selector */
p {
  color: blue;
}

/* Class Selector */
.className {
  font-size: 16px;
}

/* ID Selector */
#uniqueID {
  text-align: center;
}

Box Model

Box Model Properties

/* Margin */
div {
  margin: 10px;
}

/* Padding */
div {
  padding: 10px;
}

/* Border */
div {
  border: 1px solid black;
}

/* Width and Height */
div {
  width: 100px;
  height: 100px;
}

Text and Font

Text Properties

/* Text Color */
p {
  color: red;
}

/* Text Alignment */
h1 {
  text-align: center;
}

/* Text Decoration */
a {
  text-decoration: none;
}

Font Properties

/* Font Size */
p {
  font-size: 16px;
}

/* Font Family */
h1 {
  font-family: Arial, sans-serif;
}

/* Font Weight */
h2 {
  font-weight: bold;
}

Positioning

Position Properties

/* Static Positioning */
div {
  position: static;
}

/* Relative Positioning */
div {
  position: relative;
  top: 10px;
}

/* Absolute Positioning */
div {
  position: absolute;
  right: 10px;
}

/* Fixed Positioning */
div {
  position: fixed;
  bottom: 0;
}

Flexbox

Common Flexbox Properties

/* Flex Container */
div {
  display: flex;
  justify-content: center; /* Align items horizontally */
  align-items: center; /* Align items vertically */
}

/* Flex Item */
div {
  flex: 1; /* Grow to fill available space */
}

Grid

Common Grid Properties

/* Grid Container */
div {
  display: grid;
  grid-template-columns: repeat(3, 1fr); /* 3 equal columns */
  gap: 10px; /* Spacing between grid items */
}

/* Grid Item */
div {
  grid-column: span 2; /* Span 2 columns */
}

Animation

Animation

/* Define a keyframe animation */
@keyframes example {
    from {background-color: red;}
    to {background-color: yellow;}
}

/* Apply the animation */
div {
    animation-name: example;
    animation-duration: 4s;
    animation-timing-function: ease-in-out;
    animation-delay: 1s;
    animation-iteration-count: infinite;
    animation-direction: alternate;
}

Transition

Transition

/* Apply a transition effect */
button {
    background-color: blue;
    color: white;
    transition: background-color 0.5s ease, transform 0.3s ease;
}

/* Change on hover */
button:hover {
    background-color: green;
    transform: scale(1.1);
}
